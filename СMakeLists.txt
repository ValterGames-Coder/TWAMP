cmake_minimum_required(VERSION 3.10)
project(twamp_server)

set(CMAKE_CXX_STANDARD 17)
set(CMAKE_CXX_STANDARD_REQUIRED ON)

add_executable(twamp_server main.cpp)

target_compile_options(twamp_server PRIVATE -Wall -Wextra -pedantic)
install(TARGETS twamp_server DESTINATION /usr/local/bin)
install(FILES ${CMAKE_SOURCE_DIR}/twampd.service DESTINATION /etc/systemd/system)

# Пост-инсталляционный скрипт
install(CODE [[
  message("-- Reloading systemd daemon and enabling twampd.service")
  execute_process(COMMAND systemctl daemon-reload)
  execute_process(COMMAND systemctl enable --now twampd.service)
]])
